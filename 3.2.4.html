# Loi b dng khng c CustomerID v s lng m (n hy)
df = df.dropna(subset=['CustomerID'])
df = df[df['Quantity'] > 0]

# Tnh tng tin cho mi dng (TotalSum = Quantity * UnitPrice)
df['TotalSum'] = df['Quantity'] * df['UnitPrice']

# Chuyn i InvoiceDate sang dng datetime
df['InvoiceDate'] = pd.to_datetime(df['InvoiceDate'], format='%d-%m-%Y %H:%M')
In[6]:
# 2. TO TNH NNG RFM (Recency, Frequency, Monetary)
# Ly ngy sau ngy mua hng cui cng trong data lm mc hin ti
snapshot_date = df['InvoiceDate'].max() + timedelta(days=1)
# Tng hp d liu theo CustomerID
rfm = df.groupby(['CustomerID']).agg({
    'InvoiceDate': lambda x: (snapshot_date - x.max()).days, # Recency
    'InvoiceNo': 'count',                                   # Frequency
    'TotalSum': 'sum'                                       # Monetary
})
#i tn ct 
rfm.rename(columns={'InvoiceDate': 'Recency',
                    'InvoiceNo': 'Frequency',
                    'TotalSum': 'Monetary'}, inplace=True)

# 6. GN NHN CM V T TN (VIP, TRUNG, BNH THNG)
from scipy.cluster.hierarchy import fcluster

# Ct cy  to ra 3 cm
labels = fcluster(Z, 3, criterion='maxclust')
rfm['Cluster_Labels'] = labels

# --- BT U ON T TN CHO KHCH HNG TNG PHN KHC ---
# 1. Tnh gi tr trung bnh 'Monetary' ca tng cm v sp xp t Cao -> Thp
mean_monetary = rfm.groupby('Cluster_Labels')['Monetary'].mean().sort_values(ascending=False)

# 2. To t in nh x (Mapping)
mapping_names = {
    mean_monetary.index[0]: 'Khch VIP',
    mean_monetary.index[1]: 'Khch Trung',
    mean_monetary.index[2]: 'Bnh thng'
}

# 3. To ct mi vi tn phn khc d hiu
rfm['Phn khc'] = rfm['Cluster_Labels'].map(mapping_names)

# Xem thng k trung bnh ca tng cm sau khi  t tn
print("\n>>> C IM CC CM KHCH HNG:")
# Sp xp theo Th t VIP , Trung , Thng
print(rfm.groupby('Phn khc')[['Recency', 'Frequency', 'Monetary']].mean().reindex(['Khch VIP', 'Khch Trung', 'Bnh thng']))
